        initial={{ opacity: 0 }}
        animate={{ opacity: 1 }}
        transition={{
          delay: index * 0.2,
          duration: 1.2,
          repeat: Infinity,
          repeatType: 'reverse',
        }}
      ></motion.span>
    ))}

);
